## 1. Pseudocode Standard
* 用伪代码描述将十进制转换成16进制的方法  
```
    INPUT number x
    For number x DO
        a = x divide 16
        b = the surplus
        x = a
        store b to an array
    UNTIL a = 0
    PRINT the array in inverse order
```
* C语言实现
```
    #include<stdio.h>
    #include<math.h>
    int main(){
        int num,surplus,result[100],n,x; //result[]大小可调节
        int i = 0;
        scanf("%d",&num);//input number
        do{
            surplus = num % 16;
            result[i] = surplus;//将余数储存在数组之中           
            i++;
            num = num / 16;
        }while(num > 0);
        for(n = i - 1;n >= 0;n--){
            x = result[n];
            if(x >= 0&&x <= 9)
                printf("%d",x);
            else
                printf("%c",x + 55);//将10-15转换为A-E
        } //将数组倒序输出
           
        return 0;

    }
```
